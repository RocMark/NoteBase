    // str 轉成 arr (split)

//* 解
```js
    let str = 'Hello'
    let orgArr = str.split('')
    let newArr = []
    orgArr.forEach((elem) => {
        newArr.unshift(elem)
    })
    console.log(newArr)
    let newStr = newArr.reduce((sum, elem) => sum + elem, '')
    console.log(newStr)
```
