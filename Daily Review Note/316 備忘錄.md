# forEach()
```js
    //* 基礎用法
    elems.forEach((elem) => {
        console.log(elem)
    })

    //* with Index
    function testArr(elem, index, array) {
        console.log(`第${index}個為${elem}`)
    }
    // index 2 被跳過 因為該處無item
    [2, 5, , 9].forEach(testArr)
    // log結果 a[0] = 2 // a[1] = 5 // a[3] = 9
```

# map() filter() 合併範例
```js
    let testArr = [1, 2, 3]
    const checkNum = elem => elem >= 2
    // 存取新陣列
    let newArr = testArr
        // filter 將 符合運算數的值
        .filter(checkNum)
        // 將數值*2 
        .map(val => val * 2)
    console.log(newArr) // [4,6]
```
