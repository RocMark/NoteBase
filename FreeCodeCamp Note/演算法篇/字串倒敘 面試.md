//* 個人解
```js
    function reverseString(str) {
        let orgArr = str.split('')
        let newArr = []
        orgArr.forEach((elem) => { newArr.unshift(elem) })
        let newStr = newArr.reduce((sum, elem) => sum + elem, '')
        return newStr
    }
```

    function reverseString(str) {
        return str.split('').reverse().join('')
